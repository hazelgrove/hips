% Circular pie
\newcommand{\pie}[2]{%
  \begin{tikzpicture}
    \draw (0,0) circle (#1); \fill[rotate=90] (#1,0) arc (0:#2:#1) -- (0,0) -- cycle;
  \end{tikzpicture}%
}
